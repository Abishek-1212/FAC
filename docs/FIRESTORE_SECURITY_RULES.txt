// FIRESTORE SECURITY RULES
// Copy these rules to your Firestore console under Rules tab

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection
    match /users/{userId} {
      allow read: if request.auth.uid == userId || request.auth.uid != null;
      allow write: if request.auth.uid == userId;
    }

    // Service Jobs collection
    match /service_jobs/{jobId} {
      // Admin can read all jobs
      allow read: if request.auth.token.role == 'admin';
      // Technician can read only their assigned jobs with stockAssigned: true
      allow read: if request.auth.token.role == 'technician' && resource.data.technicianId == request.auth.uid && resource.data.stockAssigned == true;
      // Admin can create and update jobs
      allow create, update: if request.auth.token.role == 'admin';
      // Technician can update their own jobs (status, completion notes)
      allow update: if request.auth.token.role == 'technician' && resource.data.technicianId == request.auth.uid;
    }

    // Job Stock Assignment collection
    match /job_stock_assignment/{assignmentId} {
      // Admin can read all assignments
      allow read: if request.auth.token.role == 'admin';
      // Technician can read their own assignments
      allow read: if request.auth.token.role == 'technician' && resource.data.technicianId == request.auth.uid;
      // Admin can create assignments
      allow create: if request.auth.token.role == 'admin';
      // Technician can update their own assignments (tracking)
      allow update: if request.auth.token.role == 'technician' && resource.data.technicianId == request.auth.uid;
      // Admin can update for verification
      allow update: if request.auth.token.role == 'admin';
    }

    // Job Completion Reports collection
    match /job_completion_reports/{reportId} {
      // Admin can read all reports
      allow read: if request.auth.token.role == 'admin';
      // Technician can read their own reports
      allow read: if request.auth.token.role == 'technician' && resource.data.technicianId == request.auth.uid;
      // Technician can create their own reports
      allow create: if request.auth.token.role == 'technician' && request.resource.data.technicianId == request.auth.uid;
    }

    // Inventory collection
    match /inventory/{productId} {
      allow read: if request.auth.uid != null;
      allow write: if request.auth.token.role == 'admin';
    }

    // Stock Transactions collection
    match /stock_transactions/{transactionId} {
      allow read: if request.auth.uid != null;
      allow create: if request.auth.uid != null;
    }

    // Invoices collection
    match /invoices/{invoiceId} {
      allow read: if request.auth.uid != null;
      allow write: if request.auth.token.role == 'admin';
    }

    // Default deny all
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
